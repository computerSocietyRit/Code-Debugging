
#include<iostream>
#include<mat.h>
using namespace std;
int main(){
int num,d=0,n,r,p=0,a=1;
char choice;
cin>>num; // cascade operator //
cin>>choice;// cascade operator //
switch(choice) // switch spelling//
{
case 'a': // ":" missing,within quotes //
n=num;
while(n==0) 
{
n=n/10;
d++;
}
n=num;
while(n!=0)
{
r=n%10;
a=a*pow(r,d); // assign symbol,power function is to be used //
n=n/10;
}
cout<<a<<"\n";  
break; // exit the loop //
case 'p': // ":" missing , within quotes// 
n=num;
while(n!=0)
{
r=n%10;
p=p+10+r; // assign symbol //
n=n/10;
continue;
}
cout<<p<<"\n";
}
}
